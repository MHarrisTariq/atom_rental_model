🎯 Price Optimization Model
def optimize_price(context, model):
best_price = context["price"]
best_revenue = 0
for p in range(80, 500, 10):
context["price"] = p
prob = model.predict(context)
revenue = p * prob
if revenue > best_revenue:
best_revenue = revenue
best_price = p
return best_price
